  // 创建代理中间件
  const proxy = createProxyMiddleware({
    target: 'https://api.minimaxi.com',

    onError: (err, req, res) => {

    }
  });

});

const LEADERBOARD_FILE = pathModule.join(DATA_DIR, 'leaderboard.json');

async function ensureDataDir() {
  try {
    await fs.mkdir(DATA_DIR, { recursive: true });
    // 初始化排行榜文件（如果不存在）
    try {
      await fs.access(LEADERBOARD_FILE);
    } catch {
      await fs.writeFile(LEADERBOARD_FILE, JSON.stringify([]));

  } catch (error) {
    console.error('Error creating data directory:', error);

app.use((req, res, next) => {

  res.header('Access-Control-Allow-Methods', 'GET, POST, OPTIONS');

    next();

    target: 'https://dashscope.aliyuncs.com',

      '^/api': '/api/v1/services/aigc/text-generation/generation', // Qwen API 路径

    if (!playerName || !score || typeof score !== 'number' || score < 0) {

    }
    
    // 限制玩家名字长度
    const safePlayerName = playerName.toString().substring(0, 20).trim();

class GameEngine {

    });

    // 设置游戏UI元素
    this.setupLeaderboardButton();
    this.setupSubmitScoreButton();
  }

    const leaderboardBtn = document.getElementById('show-leaderboard');

        this.showLeaderboard();
      });
    }
  }

    if (submitBtn) {

      });

    try {
      const response = await fetch(this.leaderboardUrl);
      if (!response.ok) {
        throw new Error('Failed to load leaderboard');
      }
      
      const leaderboard = await response.json();
      this.displayLeaderboard(leaderboard);
    } catch (error) {
      console.error('Error loading leaderboard:', error);
      alert('无法加载排行榜，请稍后再试');
    }
  }

  displayLeaderboard(leaderboard) {
    const leaderboardContainer = document.getElementById('leaderboard-container');
    if (!leaderboardContainer) return;

    if (leaderboard.length === 0) {
      leaderboardContainer.innerHTML = '<p>暂无排行榜数据</p>';
      return;
    }

    let html = '<h3>排行榜</h3><ol>';
    leaderboard.forEach((entry, index) => {
      html += `<li>
        <span class="rank">${index + 1}</span>
        <span class="player-name">${this.escapeHtml(entry.playerName)}</span>
        <span class="score">${entry.score}</span>
      </li>`;
    });
    html += '</ol>';

    leaderboardContainer.innerHTML = html;
    leaderboardContainer.style.display = 'block';
  }

    try {
      const response = await fetch(this.leaderboardUrl, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          playerName: playerName,
          score: score,
          gameData: gameData
        })
      });

      if (!response.ok) {
        const errorData = await response.json();
        throw new Error(errorData.error || '提交失败');
      }

      const result = await response.json();
      console.log('Score submitted:', result);
      return true;
    } catch (error) {
      console.error('Error submitting score:', error);
      alert('提交分数失败: ' + error.message);
      return false;
    }
  }
